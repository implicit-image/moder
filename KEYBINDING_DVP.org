* KEYBINDING PROGRAMMER DVORAK

This is the ~moder-setup~ function for [[https://www.kaufmann.no/roland/dvorak/][Programmer Dvorak]] layout.
Add it to your configuration and call it before ~(moder-global-mode 1)~.

[[file:https://user-images.githubusercontent.com/11796018/144638747-7e522ef2-b8c6-440f-ba7a-6ce55a96ecdd.png]]

#+begin_src emacs-lisp
  (defun moder-setup ()
    (setq moder-cheatsheet-layout moder-cheatsheet-layout-dvp)
    (moder-leader-define-key
     '("?" . moder-cheatsheet))
    (moder-motion-define-key
     ;; custom keybinding for motion state
     '("<escape>" . ignore))
    (moder-normal-define-key
     '("?" . moder-cheatsheet)
     '("*" . moder-expand-0)
     '("=" . moder-expand-9)
     '("!" . moder-expand-8)
     '("[" . moder-expand-7)
     '("]" . moder-expand-6)
     '("{" . moder-expand-5)
     '("+" . moder-expand-4)
     '("}" . moder-expand-3)
     '(")" . moder-expand-2)
     '("(" . moder-expand-1)
     '("1" . digit-argument)
     '("2" . digit-argument)
     '("3" . digit-argument)
     '("4" . digit-argument)
     '("5" . digit-argument)
     '("6" . digit-argument)
     '("7" . digit-argument)
     '("8" . digit-argument)
     '("9" . digit-argument)
     '("0" . digit-argument)
     '("-" . negative-argument)
     '(";" . moder-reverse)
     '("," . moder-inner-of-thing)
     '("." . moder-bounds-of-thing)
     '("<" . moder-beginning-of-thing)
     '(">" . moder-end-of-thing)
     '("a" . moder-append)
     '("A" . moder-open-below)
     '("b" . moder-back-word)
     '("B" . moder-back-symbol)
     '("c" . moder-change)
     '("d" . moder-delete)
     '("D" . moder-backward-delete)
     '("e" . moder-line)
     '("E" . moder-goto-line)
     '("f" . moder-find)
     '("g" . moder-cancel-selection)
     '("G" . moder-grab)
     '("h" . moder-left)
     '("H" . moder-left-expand)
     '("i" . moder-insert)
     '("I" . moder-open-above)
     '("j" . moder-join)
     '("k" . moder-kill)
     '("l" . moder-till)
     '("m" . moder-mark-word)
     '("M" . moder-mark-symbol)
     '("n" . moder-next)
     '("N" . moder-next-expand)
     '("o" . moder-block)
     '("O" . moder-to-block)
     '("p" . moder-prev)
     '("P" . moder-prev-expand)
     '("q" . moder-quit)
     '("r" . moder-replace)
     '("R" . moder-swap-grab)
     '("s" . moder-search)
     '("t" . moder-right)
     '("T" . moder-right-expand)
     '("u" . moder-undo)
     '("U" . moder-undo-in-selection)
     '("v" . moder-visit)
     '("w" . moder-next-word)
     '("W" . moder-next-symbol)
     '("x" . moder-save)
     '("X" . moder-sync-grab)
     '("y" . moder-yank)
     '("z" . moder-pop-selection)
     '("'" . repeat)
     '("<escape>" . ignore)))
#+end_src
