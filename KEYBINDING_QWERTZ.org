* KEYBINDINGS QWERTZ

This is the example ~moder-setup~ function for Qwertz layout.
Add it to your configuration and call it before ~(moder-global-mode 1)~.

NOTE:
ex means this command will expand current region.
←thing→ (inner), [thing] (bounds), ←thing (begin) and thing→ (end) require a THING as input:
    round ->  f    square ->  d     curly ->  s     angle ->  a    string ->  r
paragraph ->  v      line ->  c    buffer ->  x

#+begin_src emacs-lisp
  (defun moder-setup ()
    (setq moder-cheatsheet-physical-layout moder-cheatsheet-physical-layout-iso)
    (setq moder-cheatsheet-layout moder-cheatsheet-layout-qwertz)

    (moder-thing-register 'angle
                         '(pair (";") (":"))
                         '(pair (";") (":")))

    (setq moder-char-thing-table
          '((?f . round)
            (?d . square)
            (?s . curly)
            (?a . angle)
            (?r . string)
            (?v . paragraph)
            (?c . line)
            (?x . buffer)))

    (moder-leader-define-key
      ;; Use SPC (0-9) for digit arguments.
      '("1" . moder-digit-argument)
      '("2" . moder-digit-argument)
      '("3" . moder-digit-argument)
      '("4" . moder-digit-argument)
      '("5" . moder-digit-argument)
      '("6" . moder-digit-argument)
      '("7" . moder-digit-argument)
      '("8" . moder-digit-argument)
      '("9" . moder-digit-argument)
      '("0" . moder-digit-argument)
      '("-" . moder-keypad-describe-key)
      '("_" . moder-cheatsheet))

    (moder-normal-define-key
      ;; expansion
      '("0" . moder-expand-0)
      '("1" . moder-expand-1)
      '("2" . moder-expand-2)
      '("3" . moder-expand-3)
      '("4" . moder-expand-4)
      '("5" . moder-expand-5)
      '("6" . moder-expand-6)
      '("7" . moder-expand-7)
      '("8" . moder-expand-8)
      '("9" . moder-expand-9)
      '("ä" . moder-reverse)

      ;; movement
      '("i" . moder-prev)
      '("k" . moder-next)
      '("j" . moder-left)
      '("l" . moder-right)

      '("z" . moder-search)
      '("-" . moder-visit)

      ;; expansion
      '("I" . moder-prev-expand)
      '("K" . moder-next-expand)
      '("J" . moder-left-expand)
      '("L" . moder-right-expand)

      '("u" . moder-back-word)
      '("U" . moder-back-symbol)
      '("o" . moder-next-word)
      '("O" . moder-next-symbol)

      '("a" . moder-mark-word)
      '("A" . moder-mark-symbol)
      '("s" . moder-line)
      '("S" . moder-goto-line)
      '("w" . moder-block)
      '("q" . moder-join)
      '("g" . moder-grab)
      '("G" . moder-pop-grab)
      '("m" . moder-swap-grab)
      '("M" . moder-sync-grab)
      '("p" . moder-cancel-selection)
      '("P" . moder-pop-selection)

      '("x" . moder-till)
      '("y" . moder-find)

      '("," . moder-beginning-of-thing)
      '("." . moder-end-of-thing)
      '(";" . moder-inner-of-thing)
      '(":" . moder-bounds-of-thing)

      ;; editing
      '("d" . moder-kill)
      '("f" . moder-change)
      '("t" . moder-delete)
      '("c" . moder-save)
      '("v" . moder-yank)
      '("V" . moder-yank-pop)

      '("e" . moder-insert)
      '("E" . moder-open-above)
      '("r" . moder-append)
      '("R" . moder-open-below)

      '("h" . undo-only)
      '("H" . undo-redo)

      '("b" . open-line)
      '("B" . split-line)

      '("ü" . indent-rigidly-left-to-tab-stop)
      '("+" . indent-rigidly-right-to-tab-stop)

      ;; ignore escape
      '("<escape>" . ignore)))
#+end_src
