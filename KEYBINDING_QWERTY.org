* KEYBINDING QWERTY

This is the example ~moder-setup~ function for Qwerty layout.
Add it to your configuration and call it before ~(moder-global-mode 1)~.

[[file:https://user-images.githubusercontent.com/11796018/144638475-160ff071-7f66-4155-b8b6-28f3db15afd7.png]]

#+begin_src emacs-lisp
  (defun moder-setup ()
    (setq moder-cheatsheet-layout moder-cheatsheet-layout-qwerty)
    (moder-motion-define-key
     '("j" . moder-next)
     '("k" . moder-prev)
     '("<escape>" . ignore))
    (moder-leader-define-key
     ;; Use SPC (0-9) for digit arguments.
     '("1" . moder-digit-argument)
     '("2" . moder-digit-argument)
     '("3" . moder-digit-argument)
     '("4" . moder-digit-argument)
     '("5" . moder-digit-argument)
     '("6" . moder-digit-argument)
     '("7" . moder-digit-argument)
     '("8" . moder-digit-argument)
     '("9" . moder-digit-argument)
     '("0" . moder-digit-argument)
     '("/" . moder-keypad-describe-key)
     '("?" . moder-cheatsheet))
    (moder-normal-define-key
     '("0" . moder-expand-0)
     '("9" . moder-expand-9)
     '("8" . moder-expand-8)
     '("7" . moder-expand-7)
     '("6" . moder-expand-6)
     '("5" . moder-expand-5)
     '("4" . moder-expand-4)
     '("3" . moder-expand-3)
     '("2" . moder-expand-2)
     '("1" . moder-expand-1)
     '("-" . negative-argument)
     '(";" . moder-reverse)
     '("," . moder-inner-of-thing)
     '("." . moder-bounds-of-thing)
     '("[" . moder-beginning-of-thing)
     '("]" . moder-end-of-thing)
     '("a" . moder-append)
     '("A" . moder-open-below)
     '("b" . moder-back-word)
     '("B" . moder-back-symbol)
     '("c" . moder-change)
     '("d" . moder-delete)
     '("D" . moder-backward-delete)
     '("e" . moder-next-word)
     '("E" . moder-next-symbol)
     '("f" . moder-find)
     '("g" . moder-cancel-selection)
     '("G" . moder-grab)
     '("h" . moder-left)
     '("H" . moder-left-expand)
     '("i" . moder-insert)
     '("I" . moder-open-above)
     '("j" . moder-next)
     '("J" . moder-next-expand)
     '("k" . moder-prev)
     '("K" . moder-prev-expand)
     '("l" . moder-right)
     '("L" . moder-right-expand)
     '("m" . moder-join)
     '("n" . moder-search)
     '("o" . moder-block)
     '("O" . moder-to-block)
     '("p" . moder-yank)
     '("q" . moder-quit)
     '("Q" . moder-goto-line)
     '("r" . moder-replace)
     '("R" . moder-swap-grab)
     '("s" . moder-kill)
     '("t" . moder-till)
     '("u" . moder-undo)
     '("U" . moder-undo-in-selection)
     '("v" . moder-visit)
     '("w" . moder-mark-word)
     '("W" . moder-mark-symbol)
     '("x" . moder-line)
     '("X" . moder-goto-line)
     '("y" . moder-save)
     '("Y" . moder-sync-grab)
     '("z" . moder-pop-selection)
     '("'" . repeat)
     '("<escape>" . ignore)))
#+end_src
